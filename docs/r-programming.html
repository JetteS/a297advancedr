3.1.3 Character operations

paste("I", "am", "me")
#> [1] "I am me"
paste0("test", 0)
#> [1] "test0"
paste0("PC", 1:10)
#>  [1] "PC1"  "PC2"  "PC3"  "PC4"  "PC5"  "PC6"  "PC7"  "PC8"  "PC9"  "PC10"
me <- "Florian"
glue::glue("I am {me}")  ## not in base, but SO useful
#> I am Florian
(x <- list.files(pattern = "\\.Rmd$"))
#> [1] "01-intro.Rmd"       "good-practices.Rmd" "index.Rmd"         
#> [4] "intro.Rmd"          "rprog.Rmd"          "rstudio-git.Rmd"
sub("\\.Rmd$", ".pdf", x)
#> [1] "01-intro.pdf"       "good-practices.pdf" "index.pdf"         
#> [4] "intro.pdf"          "rprog.pdf"          "rstudio-git.pdf"
(y <- sample(letters[1:4], 10, replace = TRUE))
#>  [1] "d" "c" "a" "c" "b" "d" "b" "d" "d" "d"
match(y, letters[1:4])
#>  [1] 4 3 1 3 2 4 2 4 4 4
y %in% letters[1:2]
#>  [1] FALSE FALSE  TRUE FALSE  TRUE FALSE  TRUE FALSE FALSE FALSE
rle(rep(1:4, 4:1))
#> Run Length Encoding
#>   lengths: int [1:4] 4 3 2 1
#>   values : int [1:4] 1 2 3 4
split(1:12, rep(letters[1:3], 4))
#> $a
#> [1]  1  4  7 10
#> 
#> $b
#> [1]  2  5  8 11
#> 
#> $c
#> [1]  3  6  9 12
intersect(letters[1:4], letters[3:5])
#> [1] "c" "d"
union(letters[1:4], letters[3:5])
#> [1] "a" "b" "c" "d" "e"
setdiff(letters[1:4], letters[3:5])
#> [1] "a" "b"

3.1.4 Logical operators

tryCatch(TRUE | stop("will go there"), error = function(e) print(e))
#> <simpleError in doTryCatch(return(expr), name, parentenv, handler): will go there>
TRUE || stop("won't go there")
#> [1] TRUE
c(TRUE, FALSE, TRUE, TRUE) & c(FALSE, TRUE, TRUE, TRUE) 
#> [1] FALSE FALSE  TRUE  TRUE
c(TRUE, FALSE, TRUE, TRUE) && c(FALSE, TRUE, TRUE, TRUE)  ## no warning!!
#> [1] FALSE
(x <- rnorm(10))
#>  [1] -0.2970153  1.1857493 -0.1217418 -0.3689861  1.2117053 -0.5041963
#>  [7]  0.6175849 -0.2352154  0.8074548 -0.3005149
ifelse(x > 0, x, -x)
#>  [1] 0.2970153 1.1857493 0.1217418 0.3689861 1.2117053 0.5041963 0.6175849
#>  [8] 0.2352154 0.8074548 0.3005149

Beware with ifelse() (learn more there), for example

ifelse(FALSE, 0, 1:5)
#> [1] 1
`if`(FALSE, 0, 1:5)
#> [1] 1 2 3 4 5
